 42        '''Connect to a database, creating it if it doesn't exist (in the case of sqlite).'''
 43        engine, meta = cls.new_sqlalchemy_engine(target=target, dialect=dialect, echo=echo, **engine_kwargs)

 45            target=target,
 46            dialect=dialect,
 47            engine=engine,
 48            metadata=meta,

 53        '''Raise exception if the target database does not exist.'''
 54        if dialect.startswith('sqlite') and target != ':memory:':
 55            if new_db and os.path.exists(target):
 56                raise FileExistsError(f'Database {target} already exists. Call connect_existing() instead.')
 57            elif not new_db and not os.path.exists(target):
 58                raise FileNotFoundError(f'Database {target} does not exist yet. Call connect_new() instead.')

 63        meta = sqlalchemy.MetaData()
 64        return engine, meta
