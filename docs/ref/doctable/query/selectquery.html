 29            cols: typing.List[typing.Union[str, sqlalchemy.Column]] = None, 
 30            chunksize: int = 100, 
 31            **kwargs
 32        ) -> typing.Iterable[DocTableSchema]:

 50    def select_chunks(self, 
 51            cols: typing.List[typing.Union[str, sqlalchemy.Column]] = None, 
 52            chunksize: int = 100, 
 53            limit: int = None, 
 54            raw_result: bool = False, 
 55            **kwargs,
 56        ) -> typing.Iterable[typing.List[DocTableSchema]]:
 57        ''' Performs select while querying only a subset of the results at a time.
 58        Args:
 59            cols (col name(s) or sqlalchemy object(s)): columns to query
 60            chunksize (int): size of individual queries to be made. Will
 61                load this number of rows into memory before yielding.
 62            limit (int): maximum number of rows to retrieve. Because 
 63                the limit argument is being used internally to limit data
 64                to smaller chunks, use this argument instead. Internally,
 65                this function will load a maximum of limit + chunksize 
 66                - 1 rows into memory, but yields only limit.
 67        Yields:
 68            result: chunked rows.
 69        '''
 70        select_func = self.select_raw if raw_result else self.select
 71        
 72        offset = 0
 73        while True:
 74            
 75            rows = select_func(cols, offset=offset, limit=chunksize, **kwargs)
 76            chunk = rows[:limit-offset] if limit is not None else rows
 77            
 78            yield chunk
 79            
 80            offset += len(rows)
 81            
 82            if (limit is not None and offset >= limit) or len(rows) == 0:
 83                break

328    @staticmethod
329    def select_query(
330        cols: typing.List[sqlalchemy.Column],
331        where: sqlalchemy.sql.expression.BinaryExpression = None,
332        orderby: typing.Union[sqlalchemy.Column, typing.List[sqlalchemy.Column]] = None,
333        groupby: typing.Union[sqlalchemy.Column, typing.List[sqlalchemy.Column]] = None,
334        limit: int = None,
335        wherestr: str = None,
336        offset: int = None,
337    ) -> sqlalchemy.sql.Select:
338        '''Build and exectute select query given all the conditionals provided as parameters.'''
339        
340        q: sqlalchemy.sql.Select = sqlalchemy.sql.select(cols)
341        
342        if where is not None:
343            q = q.where(where)
344        
345        if wherestr is not None:
346            q = q.where(sqlalchemy.text(f'({wherestr})'))
347        
348        if orderby is not None:
349            if is_sequence(orderby):
350                q = q.order_by(*orderby)
351            else:
352                q = q.order_by(orderby)
353        
354        if groupby is not None:
355            if is_sequence(groupby):
356                q = q.group_by(*groupby)
357            else:
358                q = q.group_by(groupby)
359            
360        if limit is not None:
361            q = q.limit(limit)
362            
363        if offset is not None:
364            q = q.offset(offset)
365            
366        return q

370    def parse_input_cols(self, cols: typing.List[typing.Union[str, sqlalchemy.Column]]) -> typing.List[sqlalchemy.Column]:
371        '''Pass variable passed to cols.'''        
372        if cols is None:
373            cols = list(self.dtab.columns)
374        else:
375            if not is_sequence(cols):
376                raise TypeError('cols argument should be a list of columns.')
377
378            cols = [self.dtab.col(c) if isinstance(c,str) else c for c in cols]
379        
380        return cols
381    
382    def parse_input_col(self, col: typing.Union[str, sqlalchemy.Column]) -> typing.List[sqlalchemy.Column]:
383        if is_sequence(col):
384            raise TypeError('col argument should be single column.')
385        
386        use_col = self.dtab.col(col) if isinstance(col,str) else col
387        return [use_col]
