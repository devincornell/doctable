171    def get_childs(self, dep: str=None, pos: str=None, matchfunc:typing.Callable=None):
172        ''' Get children with the specified relations.
173        Args:
174            dep (sequence or string): dependency relations to match on.
175            pos (sequence or string): pos to match on.
176            matchfunc (function or None): additional custom matching function.
177        '''
178        if isinstance(dep, str):
179            dep = set([dep])
180
181        if isinstance(pos, str):
182            pos = set([pos])
183
184        childs = list()
185        for c in self.childs:
186            if (dep is None or c.dep in dep): # nested to make more readable?
187                if (pos is None or c.pos in pos):
188                    if (matchfunc is None or matchfunc(c)):
189                        childs.append(c)
190        return childs
191
192    def get_child(self, *args, allow_multiple: bool=False, **kwargs):
193        ''' Get first child with the given dependency relation.
194        Args:
195            *args: passed to get_childs
196            allow_multiple: don't allow 
197            **kwargs: passed to get_childs
198        Raises:
199            ValueError when the token has more than one dependency with the 
200                given relation.
201        '''
202        childs = self.get_childs(*args, **kwargs)
203        if len(childs) == 1 or (len(childs) and allow_multiple):
204            return childs[0]
205        elif not len(childs):
206            return None
207        else:
208            raise ValueError(f'There is more than one dependency matching {args}, {kwargs}.')
209    
210    def get_preps(self, as_str:bool=False):
211        ''' Gets chained prepositional phrases starting at the current token.
212        Returns:
213            tuple of prep, pobj.
214        '''
215        preps = list()
216        for prep in self.get_childs({'prep', 'dative'}):
217            if as_str:
218                pairs = (prep.t, [p.t for p in prep.get_childs('pobj') if p is not None])
219            else:
220                pairs = (prep, [p for p in prep.get_childs('pobj') if p is not None])
221            preps.append(pairs)
222        
223        return preps
224    
225    def bubble_accum(self, func: typing.Callable):
226        ''' Bubble up results into a list.
227        Args:
228            func: function that accepts a Token and returns a list
229                of results that will be concatenated at each level
230                of the tree.
231        Example:
232            `self.bubble_accum(lambda n: [n])` would return a list
233                of (unordered) tokens in the tree below the given 
234                node.
235        '''
236        aggregated_list = func(self)
237        for child in self.childs:
238            aggregated_list += child.bubble_accum(func)
239        return aggregated_list
