 42    def get_parsetrees(cls, doc, parse_tok_func=None, userdata_map=dict(), merge_ents=False, 
 43            spacy_ngram_matcher=None, merge_noun_chunks=False):
 44        '''Extracts parsetree from spacy doc objects.
 45        Args:
 46            doc (spacy.Doc object): doc to generate parsetree from.
 47            parse_tok_func (func): function used to convert token to 
 48                a string representation. Usually a lambda function 
 49                wrapping some variant of self.parse_tok().
 50            userdata_map (dict<str->func>): attribute to function 
 51                mapping. Functions take a token and output a property
 52                that will be stored in each parsetree node.
 53            merge_ents (bool): merge multi-word entities.
 54            spacy_ngram_matcher (Spacy Matcher): used to create ngrams
 55                with Spacy. Powerful wildcards etc.
 56            merge_noun_chunks (bool): merge noun chunks or not.
 57        '''
 58        if parse_tok_func is None:
 59            parse_tok_func = cls.parse_tok
 60        
 61        # apply ngram merges to doc object (permanently modifies doc object)
 62        cls.apply_doc_transform(
 63            doc, 
 64            merge_ents=merge_ents, 
 65            spacy_ngram_matcher=spacy_ngram_matcher, 
 66            merge_noun_chunks=merge_noun_chunks
 67        )
 68        
 69        sent_trees = [
 70            ParseTree(sent.root, parse_tok_func, userdata_map=userdata_map)
 71            for sent in doc.sents
 72        ]
 73        return sent_trees

 78    def tokenize_doc(cls, doc, split_sents=False, merge_ents=False, merge_noun_chunks=False, 
 79        ngrams=list(), spacy_ngram_matcher=None, ngram_sep=' ', use_tok_func=None, 
 80        parse_tok_func=None):
 81        '''Parse spacy doc object.
 82        Args:
 83            split_sents (bool): parse into list of sentence tokens using doc.sents.
 84            merge_ents (bool): merge multi_word entities into same token.
 85            ngrams (iter<iter<str>>): iterable of token tuples to merge after parsing.
 86            spacy_ngram_matcher (spacy Matcher): matcher object to use on the spacy doc.
 87                Normally will create using spacy.Matcher(nlp.vocab), see more details
 88                at https://spacy.io/usage/rule-based-matching And also note that the 
 89                nlp object must be the one used for parsing.
 90            use_tok_func (func): func used to decide to keep func or not. Default is
 91                cls.use_tok().
 92            parse_tok_func (func): func used to parse tokens. By default uses 
 93                cls.parse_tok().
 94        '''
 95        
 96        # NOTE! These need to be under two separate retokenize() blocks
 97        #     because retokenizing only occurs after the block and sometimes
 98        #     custom matches (provided via matcher) overlap with ents. Easiest
 99        #     way is to do custom first, then non-overlapping.
100        cls.apply_doc_transform(doc, 
101                merge_ents=merge_ents, 
102                spacy_ngram_matcher=spacy_ngram_matcher,
103                merge_noun_chunks=merge_noun_chunks
104            )
105                        
106        if parse_tok_func is None:
107            parse_tok_func = cls.parse_tok
108        if use_tok_func is None:
109            use_tok_func = cls.use_tok
110            
111        # sentence parsing mode
112        if split_sents:
113            sents = [
114                [parse_tok_func(tok) 
115                 for tok in sent if use_tok_func(tok)] 
116                for sent in doc.sents
117            ]
118            
119            if len(ngrams) > 0:
120                sents = [cls.merge_ngrams(sent, ngrams, ngram_sep=ngram_sep) for sent in sents]
121            
122            return sents
123        
124        # doc parsing mode
125        else:
126            toks = [parse_tok_func(tok) 
127                    for tok in doc if use_tok_func(tok)]
128            
129            if len(ngrams) > 0:
130                toks = cls.merge_ngrams(toks, ngrams, ngram_sep=ngram_sep)
131            return toks

134    def parse_tok(tok, num_replacement=None, digit_replacement=None, lemmatize=False, normal_convert=None, 
135            format_ents=False, ent_convert=None):
136        '''Convert spacy token object to string.
137        Args:
138            tok (spacy token or span): token object to convert to string.
139            replace_num (str/None): Replace number following tok.like_num (includes "five", 
140                or 5) with a special token (i.e. __NUM__). None means no replacement.
141            replace_digit (str/None): Replace digit meeting tok.is_digit with special token. 
142                Only used when replace_num is None.
143            lemmatize (bool): return lemma instead of full word.
144            normal_convert (func): custom conversion function to happen as last step
145                for non-entities. This way can keep all other functionality.
146            format_ents (bool): Replace whitespace with space and capitalize first 
147                letter of ents.
148            ent_convert (func): custom conversion function to happen as last step
149                for entities. This way can keep all other functionality.
150        '''
151        
152        # catch issues with 
153        
154        # replace characters
155        if digit_replacement is not None and tok.is_digit:
156            return digit_replacement
157        if num_replacement is not None and tok.like_num:
158            return num_replacement
159        
160        if tok.ent_type_ == '': # non-entity token
161            if lemmatize:
162                return tok.lemma_.lower().strip()
163            elif normal_convert is not None:
164                return normal_convert(tok)
165            else:
166                return tok.text.lower().strip()
167        
168        else: # token is entity
169            if ent_convert is not None:
170                return ent_convert(tok)
171            elif format_ents:
172                return ' '.join([t.strip().lower() for t in tok.text.split()]).capitalize()
173            else:
174                return tok.text.strip()

176    @staticmethod
177    def use_tok(tok, filter_whitespace=True, filter_punct=False, filter_stop=False, filter_digit=False, filter_num=False, filter_all_ents=False, filter_ent_types=tuple()):
178        '''Decide to use token or not (can be overridden).
179        Args:
180            no_whitespace (bool): exclude whitespace.
181            no_punct (bool): exclude punctuation.
182            no_num (bool): exclude numbers using tok.is_num.
183            no_digit (bool): exclude digits using tok.is_digit.
184            no_stop (bool): exclude stopwords.
185        '''
186        do_use_tok = True
187        if filter_whitespace:
188            do_use_tok = do_use_tok and (not tok.is_space and len(tok.text.strip()) > 0)
189        
190        if filter_punct:
191            do_use_tok = do_use_tok and not tok.is_punct
192            
193        if filter_stop:
194            do_use_tok = do_use_tok and not tok.is_stop
195            
196        if filter_digit:
197            do_use_tok = do_use_tok and not tok.is_digit
198            
199        if filter_num:
200            do_use_tok = do_use_tok and not tok.like_num
201        
202        if filter_all_ents:
203            do_use_tok = do_use_tok and tok.ent_type_ == ''
204            
205        if len(filter_ent_types) > 0:
206            do_use_tok = do_use_tok and tok.ent_type_ not in filter_ent_types
207            
208        return do_use_tok

211    @staticmethod
212    def apply_doc_transform(doc, merge_ents=True, spacy_ngram_matcher=None, merge_noun_chunks=False):
213        '''Apply merges to doc object including entities, normal ngrams, and noun chunks.'''
214        if spacy_ngram_matcher is not None:
215            # merge custom matches
216            with doc.retokenize() as retokenizer:
217                for match_id, start, end in spacy_ngram_matcher(doc):
218                    retokenizer.merge(doc[start:end])
219        if merge_ents:
220            # merge entities
221            with doc.retokenize() as retokenizer:
222                for ent in doc.ents:
223                    retokenizer.merge(ent)
224        if merge_noun_chunks:
225            # merge entities
226            with doc.retokenize() as retokenizer:
227                for nc in doc.noun_chunks:
228                    retokenizer.merge(nc)
229        return doc # just in case user tries to assign

231    @staticmethod
232    def merge_ngrams(toks, ngrams, ngram_sep=' '):
233        '''Merges manually specified consecutive tokens into single tokens.
234        '''
235        new_toks = list()
236        ngram_starts = [ng[0] for ng in ngrams] # first word of every ngram
237        i = 0
238        while i < len(toks):
239            if toks[i] in ngram_starts: # match is possible
240                found_match = False
241                for ng in ngrams:
242                    zip_rng = zip(range(len(ng)), range(i,len(toks)))
243                    if all([ng[j]==toks[k] for j,k in zip_rng]):
244                        new_toks.append(ngram_sep.join(ng))
245                        i += len(ng)
246                        found_match = True
247                        break
248                if not found_match:
249                    new_toks.append(toks[i])
250                    i += 1
251            else: # match is impossible
252                new_toks.append(toks[i])
253                i += 1
254        return new_toks

258    @classmethod
259    def parse_text_chunks(cls, text, nlp, parse_funcs={}, doc_transform=None,
260                             chunk_sents=1000, split_re=r'([\?\!\.]+)(?=[\WA-Z])'):
261        '''Parse text in paragraph by sentences.
262        Args:
263            text (str): (preprocessed) text document to parse
264            nlp (spaCy parser): with .pipe() method to parse documents
265            parse_funcs (list<func>): convert doc to 
266            chunk_sents (int): number of sentences used in each chunk to 
267                be parsed. Max size for single spacy doc is 1 million 
268                chars. If av num chars per sent is 75-100, a size of
269                3000 means each chunk will have approx 300k characters.
270                Hopefully, on average small enough. Larger value means
271                more memory but faster processing.
272        '''
273        if doc_transform is None:
274            doc_transform = lambda x: x
275        
276        # split into paragraphs and chunks
277        text_chunks = cls._split_texts(text, chunk_sents, split_re)
278        
279        # flatten texts into single list and record paragraph,chunk ids
280        parids = [(i,ch) for i, par in enumerate(text_chunks) for ch in par]
281        parids, allchunks = list(zip(*parids))
282                
283        # parse each document, store in structure
284        last_idx = 0 # i is paragraph, j is chunk
285        parsed_par_chunks = [[]] # list of par->chunks
286        for idx, doc in zip(parids, nlp.pipe(allchunks)):
287            doc = doc_transform(doc)
288            parsed = {k:v(doc) for k,v in parse_funcs.items()}
289            #print(idx)
290            if idx != last_idx:
291                parsed_par_chunks.append([])
292                #print('adding el', idx)
293            parsed_par_chunks[-1].append(parsed)
294            last_idx = idx
295            del doc
296        
297        return parsed_par_chunks

299    @staticmethod
300    def _split_texts(text, chunk_sents, split_re):
301        sents = [s for s in re.split(split_re, text) if len(s.split())>0]
302        n_chunks = math.ceil(len(sents)/(2*chunk_sents))
303        textblocks = [''.join(sents[i*chunk_sents*2:(i+1)*chunk_sents*2]) 
304                        for i in range(n_chunks)]
305        return textblocks

309    @classmethod
310    def distribute_parse(cls, texts, spacynlp, parsefunc=None, preprocessfunc=None, 
311        dt_inst=None, paragraph_sep=None, workers=None):
312        '''Distributes text parsing across multiple processes in chunks.
313        Args:
314            texts (list): list of raw texts to process
315            spacynlp (spacy nlp object): for applying .pipe() to doc chunks
316            parsefunc (func): convert spacy doc object to storage represenation
317                By defaut uses DocParser.tokenize_doc().
318            preprocessfunc (func): process text before parsing. Uses 
319                DocParser.preprocess() by default.
320            dt_inst (doctable instance): if included, will pass to parsefunc as
321                second argument. Usually to do this, you'll create a function
322                which calls some overloaded version of .tokenize_doc() and then
323                calls dt_inst.insert() to place in database.
324            paragraph_sep (str or None): if defined, will distribute parsing across
325                paragraphs and leave paragraph structure of docs in nested lists.
326            workers (int): number of processes to create.
327        Returns:
328            output of parsing
329        '''
330        
331        if dt_inst is not None and parsefunc is None:
332            raise ValueError('When dt_inst is provided, parsefunc must be provided '
333                'to insert elements into database after parsing.')
334        
335        # default parse functions
336        if preprocessfunc is None:
337            preprocessfunc = cls.preprocess
338        if parsefunc is None:
339            parsefunc = cls.tokenize_doc
340        
341        
342        # split into paragraphs
343        if paragraph_sep is not None:
344            texts, ind = list(zip(*[(par.strip(),i) for i,text in enumerate(texts) 
345                            for par in text.split(paragraph_sep)]))
346        
347        # perform actual parsing
348        thread_args = (spacynlp, parsefunc, preprocessfunc, dt_inst)
349        #parsed = cls.distribute_chunks(cls._distribute_parse_thread, texts, 
350        #                               *thread_args, workers=workers)
351        with Distribute(workers) as d:
352            parsed = d.map_insert(cls._distribute_parse_thread, texts,
353                                 *thread_args, dt_inst=dt_inst)
354            
355        # fold paragraphs back into document list
356        if paragraph_sep is not None:
357            parsed_docs = list()
358            last_i, last_ind = 0, 0
359            for i in range(len(ind)):
360                if ind[i] != last_ind:
361                    parsed_docs.append(parsed[last_i:i])
362                    last_ind = ind[i]
363                    last_i = i
364                elif i == len(ind)-1:
365                    parsed_docs.append(parsed[last_i:])
366            parsed = parsed_docs
367        
368        return parsed
