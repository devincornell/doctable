310            indices: typing.List[Index], 
311            table: sqlalchemy.Table, 
312            allow_inconsistent_schema: bool
313        ) -> None:
314        
315        new_indices = self.get_new_indices(table.name, indices, allow_inconsistent_schema)
316        
317        # used to convert strings to actual column objects
318        col_lookup = {c.name:c for c in table.columns}
319        
320        for ind in new_indices:
321            try:
322                sa_ind = ind.get_sqlalchemy_index(col_lookup)
323            except KeyError as e:
324                violators = {c for c in ind.columns if c not in col_lookup}
325                raise ColumnNotFoundError(f'The following index-associated columns were not found in '
326                    f'table "{table.name}": {violators}. Use create_indices=False to ignore '
327                    f'the creation of this index.')

329            # this will actually create the engine
330            sa_ind.create(self._engine)

332    def bind_column_methods(self, table: sqlalchemy.Table):
333        '''Binds sqlalchemy methods to column objects.'''
334        
335        for col in table.c:
336            col.max = functools.partial(sqlalchemy.sql.func.max, col)
337            col.min = functools.partial(sqlalchemy.sql.func.min, col)
338            col.count = functools.partial(sqlalchemy.sql.func.count, col)
339            col.sum = functools.partial(sqlalchemy.sql.func.sum, col)
340            col.unique = functools.partial(sqlalchemy.sql.func.unique, col)
341
342            def func_custom(attr: str):
343                '''Allows user to specify any sqlalchemy function name.'''
344                return getattr(sqlalchemy.sql.func, attr)(col)
345            col.func = func_custom

348    def check_column_consistency(
349            existing_colnames: typing.Set[str], 
350            provided_cols: typing.Set[sqlalchemy.Column],
351        ):
352        provided_colnames = set(c.name for c in provided_cols)
353        if existing_colnames != provided_colnames:
354            raise ValueError(f'Provided schema columns do not match existing '
355                f'db table names. {existing_colnames=}, {provided_colnames=} '
356                f'use allow_consistent_schema=True to ignore this error.')

358    def get_new_indices(self,
359            tabname: str,
360            indices: typing.List[Index],
361            allow_inconsistent_schema: bool,
362        ) -> typing.List[Index]:
363        inspector = self.inspector
364        exist_ind = {ind['name']:tn for tn in inspector.get_table_names() 
365                        for ind in inspector.get_indexes(tn)}
366                
367        # make sure existing indices with the same name are associated with same table
368        a = any(exist_ind[ind.name] != tabname for ind in indices if ind.name in exist_ind)
369        if not allow_inconsistent_schema and a:
370            violators = {ind.name: exist_ind[ind.name] for ind in indices 
371                if exist_ind.get(ind.name,tabname) != tabname}
372            raise ValueError(f'The following indices have already been '
373                f'assigned to another table: {violators=}. Set '
374                f'create_indices=False or allow_inconsistent_schema=True '
375                f'to ignore this error.')
376        
377        return [ind for ind in indices if ind.name not in exist_ind]

390        Args:

392            if_exists (bool): if true, won't throw exception if table doesn't exist.
393        '''
394        if isinstance(table, sqlalchemy.Table):
